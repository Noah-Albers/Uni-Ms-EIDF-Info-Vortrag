	\section{Differenzierung - Was ist das?}
	
	\begin{frame}[t]{Differenzierung - Was ist das?}{Äußere Differenzierung}
		\begin{itemize}
			\ppause
			\item Teilung in möglichst \textit{homogene} Gruppen
			\ppause
			\item Basiert oft auf:
			\ppause
			\begin{itemize}
				\item Leistung (Noten)
				\ppause
				\item Interessen (Fächer)
			\end{itemize}
			\teacher{HIER FRAGEN: Fallen euch Beispiele für äußere Diff. ein?}
			\ppause
			
			\item Mögliche Beispiele Äußerer Differenzierung
			\begin{itemize}
				\ppause
				\item Schulform entscheided – Hauptschule, Realschule, Gymnasium
				\ppause
				\item Fremdsprache
				\ppause
				\item Fachbelegung in der Oberstufe
			\end{itemize}
			\ppause
			$\Rightarrow$ Kann von der Lehrkraft kaum beeinflusst werden
		\end{itemize}



	\end{frame}


	
	\begin{frame}[t]{Differenzierung - Was ist das?}{Innere Differenzierung}
		\begin{itemize}
			\ppause
			\item Bezieht sich auf \textit{heterogene} Gruppen
			\ppause
			\item Maßnahmen die in den Gruppen getroffen werden 
			\ppause
			\begin{itemize}
				\item Auf Fähigkeiten der SuS eingehen
				\ppause
				\item Fertigkeiten der SuS eingehen 
				\ppause
				\item Den Interessen und Bedürfnisse entsprechen
				\ppause
				\item Individuelle Lernbedürfnisse und Wünsche beachten
				\ppause
			\end{itemize}
			\item Äußere Differenzierung reicht in der Regel nicht (kaum Umsetzbar)\\$\Rightarrow$ es braucht Innere Differenzierung
		\end{itemize}
	\end{frame}

	\section{Möglichkeiten für Binnendifferenzierung}
	\begin{frame}[t]{Arten von Binnendifferenzierung}
			\begin{itemize}
				\item Zusatzaufgaben
				\ppause
				\begin{itemize}
					\item[] \lowlight{Für Starke: Erweiternde fordernde Aufgaben}
					\ppause
					\item[] \lowlight{Für Schwache: Unterstützende Aufgaben zum Helfen}
					\ppause
				\end{itemize}
				\item Wahlaufgaben\\
				\ppause
				\hspace{.7cm} \lowlight{SuS können selber entscheiden welche Aufgaben sie bearbeiten}
				\ppause
				\item Gestaffelte Aufgaben\\
				\ppause
				\hspace{.7cm} \lowlight{Aufgaben gehen von leicht zu schwer.\\
					\ppause
				\hspace{.7cm} Z.b. \textit{faded worked examples}}
				\ppause
				\item Starke / Schwache Aufgaben\\
				\ppause
				\hspace{.7cm} \lowlight{Starke und Schwache SuS bekommen unterschiedliche Aufgaben}
				\ppause
				\item Diff. Gruppenarbeit\\
				\ppause
				\hspace{.7cm} \lowlight{Aufteilung in möglichst homogene Gruppen, dazu gleich mehr}
			\end{itemize}
	\end{frame}

	\section{Praxisbeispiel - Kryptographie im Unterricht}
	\begin{frame}[t]{Praxisbeispiel}{Kryptographie im Unterricht}
		
		\begin{itemize}
			\ppause
			\item Beispiel aus der LOG IN
			\ppause
			\item Unterrichtsreihe zur Kryptographie entwickelt und getestet
			\ppause
			\item Verlief in \textit{Block I} bis \textit{Block VI}
			\ppause
			\item Verschiedene Arten von Differenzierung versucht
		\end{itemize}
	\end{frame}
	
	\begin{frame}[t]{Praxisbeispiel}{Block I - Motivation}
		
		\begin{itemize}
			\ppause
			\item Ziel: Motivation für das Thema und Einordnung von Vorkentnissen
			\teacher{Hw: Fragebogen zum Selbsteinschätzen, teilen in mit/ohne Vorkentnisse}
			\ppause
			\item Sozialform: undifferenziert
		\end{itemize}
		
	\end{frame}
	
	
	\begin{frame}[t]{Praxisbeispiel - Block II }{Übersicht}
		
		\begin{itemize}
			\item Inhaltiche Ziel: Monoalphabetische Chiffrierverfahren kennenlernen\\
			\ppause
			\item Sozialform: Vorkenntnishomogene Gruppen
			\ppause
			\begin{itemize}
				\item SuS mit Vorkentnissen\\
				\ppause
				$\Rightarrow$ Entwickeln Program zur statischen Analyse von Texten
				\ppause
				\item SuS ohne Vorkentnisse\\
				\ppause
				$\Rightarrow$ Einführung im Frontalunterricht\\
				\ppause
				$\Rightarrow$ Entwickeln Program zum Ver/Entschlüssel von Texten
				\ppause
				\item Ca. $5-6$ Gruppen mit 2-3 SuS
			\end{itemize}
		\end{itemize}
		
	\end{frame}
	
	\begin{frame}[t]{Praxisbeispiel - Block II}{Fazit}
		\begin{itemize}
			\ppause
			\item Meisten Gruppen arbeiteten gut mit
			\ppause
			\begin{itemize}
				\item Diskutierten verschiedene Denkansätze
				\ppause
				\item Zuziehen der Lehrkraft bei Problemen
				\ppause
				\item Teils etwas planlos aber trotzdem gut mitgearbeitet
				\ppause
			\end{itemize}
			\item Teilweise gab es Probleme
			\ppause
			\begin{itemize}
				\item Gruppenkonsterlationen funktionierten nicht\\
				\teacher{Z.b. weil Streit oder verschiedene Chraktäre (Nicht homogen)}
				\ppause
				\item SuS fehlten und wurden nicht mehr richtig eingearbeitet
				\ppause
			\end{itemize}
			\item Gelöst durch
			\ppause
			\begin{itemize}
				\item Aufteilen der Gruppen, neu zusammensetzen
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\definecolor{darkgreen}{rgb}{0,0.7,0}
	
	\begin{frame}[t]{Praxisbeispiel - Block II}{Schülermeinungen}
		
		\begin{itemize}
			\ppause
			\item \textcolor{darkgreen}{\textbf{+}} Allgemein positiv wahrgenommen
			\ppause
			\begin{itemize}
				\item Gruppenmitglieder fragen war hilfreich
				\ppause
				\item Hat viel Spaß gemacht
				\ppause
				\item Eigenverantwortlich Dinge anwenden  \& erarbeiten
				\ppause
			\end{itemize}
			\item \textcolor{red}{\textbf{-}} Teilweise Probleme
			\ppause
			\begin{itemize}
				\item Teilweise fehlende Hilfestellungen\\
				\teacher{Hat Lehrkraft am ende selbst gesagt $\Rightarrow$ hat sich zu sehr zurückgehalten}
				\ppause
				\item (teils) unausgewogene Arbeitsverteilung\\
				\teacher{Gruppen wurden auseinandergenommen, neu verteilt}
				\ppause
				\item FOMO - Gefühl wichtige Dinge nicht mit zu bekommen weil andere Gruppen daran arbeiten
				\ppause
				\item Unklar ob oder was neu erlernt wurde
			\end{itemize}
		\end{itemize}
	\end{frame}	
	
	\begin{frame}[t]{Praxisbeispiel - Tipps und Tricks}{}
		\begin{itemize}
			\item Transparente Zielsetzung ist wichtig\\
			\ppause
			\teacher{Damit SuS damit einverstanden sind $\Rightarrow$ Steigert Motivation}
			\item Regel festlegen
			\ppause
			\begin{itemize}
				\item Für die Arbeit innerhalb der Gruppe
				\ppause
				\item Zur Kommunikation mit der Lehrkraft
				\ppause
				\item zusammen mit den SuS $\Rightarrow$ Regeln werden akzeptiert
				\ppause
			\end{itemize}
			\item Zusammensetzung der Gruppen...
			\ppause
			\begin{itemize}
				\item ... nach objektiv nachvollziehbaren Kriterien
				\ppause
				\item ... sollten Wünsche der SuS berücksichtigen
				\ppause
			\end{itemize}
			\item Gruppengrößen am Rechner $n \le 3$
			\ppause
			\item Kruzschrittige Arbeitsaufträge und Kontrollen\\
			\ppause
			$\Rightarrow$ Mehr Erfolgserlebnissen \& Lernkontrollen
		\end{itemize}
	\end{frame}
