	\section{Differenzierung - Was ist das?}
	
	\begin{frame}[t]{Differenzierung - Was ist das?}{Äußere Differenzierung}
		\begin{itemize}
			\ppause
			\item Teilung in möglichst \textit{homogene} Gruppen
			\ppause
			\item Basiert oft auf
			\ppause
			\begin{itemize}
				\item Leistung (Noten)
				\ppause
				\item Interessen (Fächer)
			\end{itemize}
			
			\teacher{HIER FRAGEN: Fallen euch Beispiele für äußere Differenzierung ein?}
			\ppause
			
			\item Mögliche Beispiele äußerer Differenzierung
			\begin{itemize}
				\ppause
				\item Schulform entscheidet – Hauptschule, Realschule, Gymnasium
				\ppause
				\item Fremdsprache
				\ppause
				\item Fachbelegung in der Oberstufe
			\end{itemize}
			
			\ppause
			$\Rightarrow$ Kann von der Lehrkraft kaum beeinflusst werden
		\end{itemize}
	\end{frame}
	
	\begin{frame}[t]{Differenzierung - Was ist das?}{Innere Differenzierung}
		\begin{itemize}
			\ppause
			\item Bezieht sich auf \textit{heterogene} Gruppen
			\ppause
			\item Maßnahmen, die in den Gruppen getroffen werden
			\ppause
			\begin{itemize}
				\item Auf Fähigkeiten der SuS eingehen
				\ppause
				\item Auf Fertigkeiten der SuS eingehen
				\ppause
				\item Den Interessen und Bedürfnissen entsprechen
				\ppause
				\item Individuelle Lernbedürfnisse und Wünsche beachten
				\ppause
			\end{itemize}
			
			\item Äußere Differenzierung in der Regel nicht (kaum Umsetzbar)\\
			\ppause
			$\Rightarrow$ es braucht Innere Differenzierung
		\end{itemize}
	\end{frame}

				\ppause
			\end{itemize}
			
			\item Im Informatikunterricht:
			\ppause
			\begin{itemize}
				\item Individuelle Vorkenntnisse (Computererfahrung, Programmierung \dots)
				\ppause

				\ppause
				\hspace{.7cm} teilweise wenig Erfahrung
				\ppause

	\section{Möglichkeiten für Binnendifferenzierung}
	
	\begin{frame}[t]{Arten von Binnendifferenzierung}
		\begin{itemize}
			\item Zusatzaufgaben
			\ppause
			\begin{itemize}
				\item[] \lowlight{Für Starke: erweiternde, herausfordernde Aufgaben}
				\ppause
				\item[] \lowlight{Für Schwache: unterstützende Aufgaben}
				\ppause
			\end{itemize}
			
			\item Wahlaufgaben\\
			\ppause
			\hspace{.7cm} \lowlight{SuS können selbst entscheiden, welche Aufgaben sie bearbeiten}
			
			\ppause
			\item Gestaffelte Aufgaben\\
			\ppause
			\hspace{.7cm} \lowlight{Aufgaben von leicht zu schwer, z.B. \textit{faded worked examples}}
			
			\ppause
			\item Leistungsdifferenzierte Aufgaben\\
			\ppause
			\hspace{.7cm} \lowlight{Starke und schwache SuS bekommen unterschiedliche Aufgaben}
			
			\ppause
			
			\item Differenzierte Gruppenarbeit\\
			\ppause
			\hspace{.7cm} \lowlight{Aufteilung in möglichst homogene Gruppen, dazu gleich mehr}
		\end{itemize}
	\end{frame}

	\section{Praxisbeispiel - Kryptographie im Unterricht}
	
	\begin{frame}[t]{Praxisbeispiel}{Kryptographie im Unterricht}
		\begin{itemize}
			\ppause
			\item Beispiel aus der LOG IN
			\ppause
			\item Unterrichtsreihe zur Kryptographie entwickelt und getestet
			\ppause
			\item Verlief in \textit{Block I} bis \textit{Block VI}
			\ppause
			\item Verschiedene Arten der Differenzierung wurden ausprobiert
		\end{itemize}
	\end{frame}
	
	\begin{frame}[t]{Praxisbeispiel}{Block I - Motivation}
		\begin{itemize}
			\ppause
			\item Ziel: Motivation für das Thema und Einordnung von Vorkenntnissen
			\teacher{HW: Fragebogen zur Selbsteinschätzung, Einteilung in mit/ohne Vorkenntnisse}
			\ppause
			\item Sozialform: undifferenziert
		\end{itemize}
	\end{frame}
	
	\begin{frame}[t]{Praxisbeispiel - Block II}{Übersicht}
		\begin{itemize}
			\item Inhaltliches Ziel: monoalphabetische Chiffrierverfahren kennenlernen\\
			\ppause
			\item Sozialform: vorkenntnishomogene Gruppen
			\ppause
			\begin{itemize}
				\item SuS mit Vorkenntnissen\\
				$\Rightarrow$ entwickeln Programm zur statischen Analyse von Texten
				\ppause
				\item SuS ohne Vorkenntnisse\\
				$\Rightarrow$ Einführung im Frontalunterricht\\
				$\Rightarrow$ entwickeln Programm zur Ver- und Entschlüsselung von Texten
				\ppause
				\item Ca. $5-6$ Gruppen mit 2-3 SuS
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}[t]{Praxisbeispiel - Block II}{Fazit}
		\begin{itemize}
			\ppause
			\item Die meisten Gruppen arbeiteten gut mit
			\ppause
			\begin{itemize}
				\item Diskutierten verschiedene Denkansätze
				\ppause
				\item Hinzuziehen der Lehrkraft bei Problemen
				\ppause
				\item Teils etwas planlos, aber trotzdem gute Mitarbeit
				\ppause
			\end{itemize}
			
			\item Teilweise gab es Probleme
			\ppause
			\begin{itemize}
				\item Gruppenkonstellationen funktionierten nicht\\
				\teacher{z. B. wegen Streit oder unterschiedlichen Charakteren (nicht homogen)}
				\ppause
				\item SuS fehlten und wurden nicht mehr richtig eingearbeitet
				\ppause
			\end{itemize}
			
			\item Gelöst durch
			\ppause
			\begin{itemize}
				\item Aufteilen der Gruppen und Neuverteilung
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\definecolor{darkgreen}{rgb}{0,0.7,0}
	
	\begin{frame}[t]{Praxisbeispiel - Block II}{Schülermeinungen}
		\begin{itemize}
			\ppause
			\item \textcolor{darkgreen}{\textbf{+}} Allgemein positiv wahrgenommen
			\ppause
			\begin{itemize}
				\item Fragen in der Gruppe waren hilfreich
				\ppause
				\item Hat viel Spaß gemacht
				\ppause
				\item Eigenverantwortliches Arbeiten
				\ppause
			\end{itemize}
			
			\item \textcolor{red}{\textbf{-}} Teilweise Probleme
			\ppause
			\begin{itemize}
				\item Teilweise fehlende Hilfestellungen\\
				\teacher{Lehrkraft hat im Nachhinein selbst gesagt, dass sie sich zu sehr zurückgehalten hat}
				\ppause
				\item (teils) unausgewogene Arbeitsverteilung\\
				\teacher{Gruppen wurden teilweise neu zusammengesetzt}
				\ppause
				\item FOMO-Gefühl, wichtige Inhalte zu verpassen
				\ppause
				\item Unklar, ob bzw. was neu gelernt wurde
			\end{itemize}
		\end{itemize}
	\end{frame}
	
	\begin{frame}[t]{Praxisbeispiel - Tipps und Tricks}
		\begin{itemize}
			\ppause
			\item Transparente Zielsetzung ist wichtig\\
			\teacher{Steigert die Motivation der SuS}
			\ppause
			
			\item Regeln festlegen
			\ppause
			\begin{itemize}
				\item für die Arbeit innerhalb der Gruppe
				\ppause
				\item für die Kommunikation mit der Lehrkraft
				\ppause
				\item gemeinsam mit den SuS $\Rightarrow$ erhöht Akzeptanz
			\end{itemize}
			
			\ppause
			\item Zusammensetzung der Gruppen
			\ppause
			\begin{itemize}
				\item nach objektiv nachvollziehbaren Kriterien
				\ppause
				\item unter Berücksichtigung der Wünsche der SuS
			\end{itemize}
			
			\ppause
			\item Gruppengröße am Rechner: $n \le 3$
			
			\ppause
			\item Kurzschrittige Arbeitsaufträge und Kontrollen\\
			$\Rightarrow$ mehr Erfolgserlebnisse \& Lernkontrollen
		\end{itemize}
	\end{frame}
